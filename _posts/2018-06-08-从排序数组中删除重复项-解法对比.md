```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var removeDuplicates = function(nums) {
    var cache = {};
    var index = 0;
    
    for (var i = 0 ; i < nums.length ; i++) {
        var n = nums[i];
        
        if (!cache[n]) {
            cache[n] = true;
            nums[index] = n;
            
            index++;
        }
    }
    
    return index;
};
```

```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */
var removeDuplicates = function(nums) {
    if(nums == null || nums.length == 0) return 0;
    if(nums.length == 1) return 1;
    const len = nums.length;
    let count = 0;
    for(let i = 1 ; i < len; i++){
        if(nums[count] != nums[i]){
            count++;
            nums[count] = nums[i];
        }
    }
    return ++count;
};
```

* 排序数组
* 不使用额外数组空间
